#include <iostream>

using namespace std;

int main()
{
    string s1 = "ABCDGH";
    string s2 = "AEDFHR";
    
    int l = 0;
    int lcs = 0;
    string lcstr = "";
    
    for (int i = 0; i < s1.length(); i++) {         // Traverse through first string
        char x = s1[i];
        for (int j = l; j < s2.length(); j++) {           // Traverse through second string
            if (x == s2[j]) {
                lcstr += x;
                lcs++;
                l = j;
            }
        }
        
    }
    
    cout << lcs << " - " << lcstr << endl;

    return 0;
}
